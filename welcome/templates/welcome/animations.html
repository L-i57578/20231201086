CSS动画代码示例

/* 淡入动画定义 */
@keyframes fadeIn {
    from {
        opacity: 0;
        transform: translateY(20px);
    }
    to {
        opacity: 1;
        transform: translateY(0);
    }
}

/* 缩放动画定义 */
@keyframes scaleUp {
    0% {
        opacity: 0;
        transform: scale(0.5);
    }
    75% {
        opacity: 0.8;
        transform: scale(1.1);
    }
    100% {
        opacity: 1;
        transform: scale(1);
    }
}
